def sigmoid(z) :
    return 1 / (1 + math.e ** (-z))

# h(x) Function. hypothesis Func.
def hypothesis(weight,X) :
    theta = np.array(weight)
    z = np.dot(X,theta)
    h = sigmoid(z)
    return h

def lossFunction(h,y) :
    # if i = 1, it occurs ZeroDivisionError ,
    # So, adjust value to 0.995
    for i in h :
        if i == 1 :
            i = 0.995
    try : 
        ret = (-y * np.log(h) - (1-y) * np.log(1-h)).mean()
    except ZeroDivisionError :
        print("Error")
    finally :
        return (-y * np.log(h) - (1-y) * np.log(1-h)).mean()

# predict Function. if h(x) returns >=0.5, set to 1. other cases, set to 0.
def predict(weight,pool) :
    m = len(pool)
    correct = 0
    for u in pool :
        x = u.get_x()
        y = u.get_y()
        l = u.label
        X = [[x,y,l]]
        predict = hypothesis(weight,X)
        if (predict >= 0.5) :
            predict = 1
        else :
            predict = 0
        
        if predict == u.label :
            correct += 1
    return correct * (1/m)
